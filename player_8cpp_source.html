367float player::fall_damage_mod() const
368{
369 if( has_effect_with_flag( "EFFECT_FEATHER_FALL" ) ) {
370 return 0.0f;
371 }
372 float ret = 1.0f;
373
374 // Ability to land properly is 2x as important as dexterity itself
375 /** @EFFECT_DEX decreases damage from falling */
376
377 /** @EFFECT_DODGE decreases damage from falling */
378 float dex_dodge = dex_cur / 2.0 + get_skill_level( skill_dodge );
379 // Penalize for wearing heavy stuff
380 const float average_leg_encumb = ( encumb( bp_leg_l ) + encumb( bp_leg_r ) ) / 2.0;
381 dex_dodge -= ( average_leg_encumb + encumb( bp_torso ) ) / 10;
382 // But prevent it from increasing damage
383 dex_dodge = std::max( 0.0f, dex_dodge );
384 // 100% damage at 0, 75% at 10, 50% at 20 and so on
385 ret *= ( 100.0f - ( dex_dodge * 4.0f ) ) / 100.0f;
386
387 if( has_trait( trait_PARKOUR ) ) {
388 ret *= 2.0f / 3.0f;
389 }
390
391 // TODO: Bonus for Judo, mutations. Penalty for heavy weight (including mutations)
392 return std::max( 0.0f, ret );
393}

396int player::impact( const int force, const tripoint &p )
397{
398 // Falls over ~30m are fatal more often than not
399 // But that would be quite a lot considering 21 z-levels in game
400 // so let's assume 1 z-level is comparable to 30 force
401
402 if( force <= 0 ) {
403 return force;
404 }
405
406 // Damage modifier (post armor)
407 float mod = 1.0f;
408 int effective_force = force;
409 int cut = 0;
410 // Percentage armor penetration - armor won't help much here
411 // TODO: Make cushioned items like bike helmets help more
412 float armor_eff = 1.0f;
413 // Shock Absorber CBM heavily reduces damage
414 const bool shock_absorbers = has_active_bionic( bionic_id( "bio_shock_absorber" ) );
415
416 // Being slammed against things rather than landing means we can't
417 // control the impact as well
418 const bool slam = p != pos();
419 std::string target_name = "a swarm of bugs";
420 Creature *critter = g->critter_at( p );
421 if( critter != this && critter != nullptr ) {
422 target_name = critter->disp_name();
423 // Slamming into creatures and NPCs
424 // TODO: Handle spikes/horns and hard materials
425 armor_eff = 0.5f; // 2x as much as with the ground
426 // TODO: Modify based on something?
427 mod = 1.0f;
428 effective_force = force;
429 } else if( const optional_vpart_position vp = g->m.veh_at( p ) ) {
430 // Slamming into vehicles
431 // TODO: Integrate it with vehicle collision function somehow
432 target_name = vp->vehicle().disp_name();
433 if( vp.part_with_feature( "SHARP", true ) ) {
434 // Now we're actually getting impaled
435 cut = force; // Lots of fun
436 }
437
438 mod = slam ? 1.0f : fall_damage_mod();
439 armor_eff = 0.25f; // Not much
440 if( !slam && vp->part_with_feature( "ROOF", true ) ) {
441 // Roof offers better landing than frame or pavement
442 // TODO: Make this not happen with heavy duty/plated roof
443 effective_force /= 2;
444 }
445 } else {
446 // Slamming into terrain/furniture
447 target_name = g->m.disp_name( p );
448 int hard_ground = g->m.has_flag( TFLAG_DIGGABLE, p ) ? 0 : 3;
449 armor_eff = 0.25f; // Not much
450 // Get cut by stuff
451 // This isn't impalement on metal wreckage, more like flying through a closed window
452 cut = g->m.has_flag( TFLAG_SHARP, p ) ? 5 : 0;
453 effective_force = force + hard_ground;
454 mod = slam ? 1.0f : fall_damage_mod();
455 if( g->m.has_furn( p ) ) {
456 // TODO: Make furniture matter
457 } else if( g->m.has_flag( TFLAG_SWIMMABLE, p ) ) {
458 // TODO: Some formula of swimming
459 effective_force /= 4;
460 }
461 }
462
463 // Rescale for huge force
464 // At >30 force, proper landing is impossible and armor helps way less
465 if( effective_force > 30 ) {
466 // Armor simply helps way less
467 armor_eff *= 30.0f / effective_force;
468 if( mod < 1.0f ) {
469 // Everything past 30 damage gets a worse modifier
470 const float scaled_mod = std::pow( mod, 30.0f / effective_force );
471 const float scaled_damage = ( 30.0f * mod ) + scaled_mod * ( effective_force - 30.0f );
472 mod = scaled_damage / effective_force;
473 }
474 }
475
476 if( !slam && mod < 1.0f && mod * force < 5 ) {
477 // Perfect landing, no damage (regardless of armor)
478 add_msg_if_player( m_warning, _( "You land on %s." ), target_name );
479 return 0;
480 }
481
482 // Shock absorbers kick in only when they need to, so if our other protections fail, fall back on them
483 if( shock_absorbers ) {
484 effective_force -= 15; // Provide a flat reduction to force
485 if( mod > 0.25f ) {
486 mod = 0.25f; // And provide a 75% reduction against that force if we don't have it already
487 }
488 if( effective_force < 0 ) {
489 effective_force = 0;
490 }
491 }
492
493 int total_dealt = 0;
494 if( mod * effective_force >= 5 ) {
495 for( int i = 0; i < num_hp_parts; i++ ) {
496 const bodypart_id bp = convert_bp( hp_to_bp( static_cast<hp_part>( i ) ) ).id();
497 const int bash = effective_force * rng( 60, 100 ) / 100;
498 damage_instance di;
499 di.add_damage( DT_BASH, bash, 0, armor_eff, mod );
500 // No good way to land on sharp stuff, so here modifier == 1.0f
501 di.add_damage( DT_CUT, cut, 0, armor_eff, 1.0f );
502 total_dealt += deal_damage( nullptr, bp, di ).total_damage();
503 }
504 }
505
506 if( total_dealt > 0 && is_player() ) {
507 // "You slam against the dirt" is fine
508 add_msg( m_bad, _( "You are slammed against %1$s for %2$d damage." ),
509 target_name, total_dealt );
510 } else if( is_player() && shock_absorbers ) {
511 add_msg( m_bad, _( "You are slammed against %s!" ),
512 target_name, total_dealt );
513 add_msg( m_good, _( "…but your shock absorbers negate the damage!" ) );
514 } else if( slam ) {
515 // Only print this line if it is a slam and not a landing
516 // Non-players should only get this one: player doesn't know how much damage was dealt
517 // and landing messages for each slammed creature would be too much
518 add_msg_player_or_npc( m_bad,
519 _( "You are slammed against %s." ),
520 _( "<npcname> is slammed against %s." ),
521 target_name );
522 } else {
523 // No landing message for NPCs
524 add_msg_if_player( m_warning, _( "You land on %s." ), target_name );
525 }
526
527 if( x_in_y( mod, 1.0f ) ) {
528 add_effect( effect_downed, rng( 1_turns, 1_turns + mod * 3_turns ) );
529 }
530
531 return total_dealt;
532}

606{

608 const bodypart_id torso_id = bodypart_id( "torso" );

899item::reload_option player::select_ammo( const item &base,
900 std::vector<item::reload_option> opts ) const
901{
902 if( opts.empty() ) {
903 add_msg_if_player( m_info, _( "Never mind." ) );
904 return item::reload_option();
905 }
906
907 uilist menu;
908 menu.text = string_format( base.is_watertight_container() ? _( "Refill %s" ) :
909 base.has_flag( "RELOAD_AND_SHOOT" ) ? _( "Select ammo for %s" ) : _( "Reload %s" ),
910 base.tname() );
911
912 // Construct item names
913 std::vector<std::string> names;
914 std::transform( opts.begin(), opts.end(),
915 std::back_inserter( names ), [&]( const item::reload_option & e ) {
916 if( e.ammo->is_magazine() && e.ammo->ammo_data() ) {
917 if( e.ammo->ammo_current() == itype_battery ) {
918 // This battery ammo is not a real object that can be recovered but pseudo-object that represents charge
919 //~ battery storage (charges)
920 return string_format( pgettext( "magazine", "%1$s (%2$d)" ), e.ammo->type_name(),
921 e.ammo->ammo_remaining() );
922 } else {
923 //~ magazine with ammo (count)
924 return string_format( pgettext( "magazine", "%1$s with %2$s (%3$d)" ), e.ammo->type_name(),
925 e.ammo->ammo_data()->nname( e.ammo->ammo_remaining() ), e.ammo->ammo_remaining() );
926 }
927 } else if( e.ammo->is_watertight_container() ||
928 ( e.ammo->is_ammo_container() && is_worn( *e.ammo ) ) ) {
929 // worn ammo containers should be named by their contents with their location also updated below
930 return e.ammo->contents.front().display_name();
931
932 } else {
933 const_cast<item_location &>( ammo_location ).make_dirty();
934 return ( ammo_location && ammo_location == e.ammo ? "* " : "" ) + e.ammo->display_name();
935 }
936 } );
937
938 // Get location descriptions
939 std::vector<std::string> where;
940 std::transform( opts.begin(), opts.end(),
941 std::back_inserter( where ), [this]( const item::reload_option & e ) {
942 bool is_ammo_container = e.ammo->is_ammo_container();
943 if( is_ammo_container || e.ammo->is_container() ) {
944 if( is_ammo_container && is_worn( *e.ammo ) ) {
945 return e.ammo->type_name();
946 }
947 return string_format( _( "%s, %s" ), e.ammo->type_name(), e.ammo.describe( &g->u ) );
948 }
949 return e.ammo.describe( &g->u );
950 } );
951
952 // Pads elements to match longest member and return length
953 auto pad = []( std::vector<std::string> &vec, int n, int t ) -> int {
954 for( const auto &e : vec )
955 {
956 n = std::max( n, utf8_width( e, true ) + t );
957 }
958 for( auto &e : vec )
959 {
960 e += std::string( n - utf8_width( e, true ), ' ' );
961 }
962 return n;
963 };
964
965 // Pad the first column including 4 trailing spaces
966 int w = pad( names, utf8_width( menu.text, true ), 6 );
967 menu.text.insert( 0, 2, ' ' ); // add space for UI hotkeys
968 menu.text += std::string( w + 2 - utf8_width( menu.text, true ), ' ' );
969
970 // Pad the location similarly (excludes leading "| " and trailing " ")
971 w = pad( where, utf8_width( _( "| Location " ) ) - 3, 6 );
972 menu.text += _( "| Location " );
973 menu.text += std::string( w + 3 - utf8_width( _( "| Location " ) ), ' ' );
974
975 menu.text += _( "| Amount " );
976 menu.text += _( "| Moves " );
977
978 // We only show ammo statistics for guns and magazines
979 if( base.is_gun() || base.is_magazine() ) {
980 menu.text += _( "| Damage | Pierce " );
981 }
982
983 auto draw_row = [&]( int idx ) {
984 const auto &sel = opts[ idx ];
985 std::string row = string_format( "%s| %s |", names[ idx ], where[ idx ] );
986 row += string_format( ( sel.ammo->is_ammo() ||
987 sel.ammo->is_ammo_container() ) ? " %-7d |" : " |", sel.qty() );
988 row += string_format( " %-7d ", sel.moves() );
989
990 if( base.is_gun() || base.is_magazine() ) {
991 const itype *ammo = sel.ammo->is_ammo_container() ? sel.ammo->contents.front().ammo_data() :
992 sel.ammo->ammo_data();
993 if( ammo ) {
994 const damage_instance &dam = ammo->ammo->damage;
995 const damage_unit &du = dam.damage_units.front();
996 if( du.damage_multiplier != 1.0f ) {
997 float dam_amt = du.amount;
998 row += string_format( "| %-3d*%3d%% ", static_cast<int>( dam_amt ),
999 clamp( static_cast<int>( du.damage_multiplier * 100 ), 0, 999 ) );
1000 } else {
1001 float dam_amt = dam.total_damage();
1002 row += string_format( "| %-8d ", static_cast<int>( dam_amt ) );
1003 }
1004 if( du.res_mult != 1.0f ) {
1005 row += string_format( "| %-3d/%3d%%",
1006 static_cast<int>( du.res_pen ), static_cast<int>( 100 * du.res_mult ) );
1007 } else {
1008 row += string_format( "| %-8d", static_cast<int>( du.res_pen ) );
1009 }
1010 } else {
1011 row += "| | ";
1012 }
1013 }
1014 return row;
1015 };
1016
1017 const ammotype base_ammotype( base.ammo_default().str() );
1018 itype_id last = uistate.lastreload[ base_ammotype ];
1019 // We keep the last key so that pressing the key twice (for example, r-r for reload)
1020 // will always pick the first option on the list.
1021 int last_key = inp_mngr.get_previously_pressed_key();
1022 bool last_key_bound = false;
1023 // This is the entry that has out default
1024 int default_to = 0;
1025
1026 // If last_key is RETURN, don't use that to override hotkey
1027 if( last_key == '\n' ) {
1028 last_key_bound = true;
1029 default_to = -1;
1030 }
1031
1032 for( auto i = 0; i < static_cast<int>( opts.size() ); ++i ) {
1033 const item &ammo = opts[ i ].ammo->is_ammo_container() ? opts[ i ].ammo->contents.front() :
1034 *opts[ i ].ammo;
1035
1036 char hotkey = -1;
1037 if( has_item( ammo ) ) {
1038 // if ammo in player possession and either it or any container has a valid invlet use this
1039 if( ammo.invlet ) {
1040 hotkey = ammo.invlet;
1041 } else {
1042 for( const auto obj : parents( ammo ) ) {
1043 if( obj->invlet ) {
1044 hotkey = obj->invlet;
1045 break;
1046 }
1047 }
1048 }
1049 }
1050 if( last == ammo.typeId() ) {
1051 if( !last_key_bound && hotkey == -1 ) {
1052 // If this is the first occurrence of the most recently used type of ammo and the hotkey
1053 // was not already set above then set it to the keypress that opened this prompt
1054 hotkey = last_key;
1055 last_key_bound = true;
1056 }
1057 if( !last_key_bound ) {
1058 // Pressing the last key defaults to the first entry of compatible type
1059 default_to = i;
1060 last_key_bound = true;
1061 }
1062 }
1063 if( hotkey == last_key ) {
1064 last_key_bound = true;
1065 // Prevent the default from being used: key is bound to something already
1066 default_to = -1;
1067 }
1068
1069 menu.addentry( i, true, hotkey, draw_row( i ) );
1070 }

1074 std::vector<item::reload_option> &opts;
1075 const std::function<std::string( int )> draw_row;
1076 int last_key;
1077 const int default_to;
1078 const bool can_partial_reload;
1079
1080 reload_callback( std::vector<item::reload_option> &_opts,
1081 std::function<std::string( int )> _draw_row,
1082 int _last_key, int _default_to, bool _can_partial_reload ) :
1083 opts( _opts ), draw_row( _draw_row ),
1084 last_key( _last_key ), default_to( _default_to ),
1085 can_partial_reload( _can_partial_reload )
1086 {}
1087
1088 bool key( const input_context &, const input_event &event, int idx, uilist *menu ) override {
1089 auto cur_key = event.get_first_input();
1090 if( default_to != -1 && cur_key == last_key ) {
1091 // Select the first entry on the list
1092 menu->ret = default_to;
1093 return true;
1094 }
1095 if( idx < 0 || idx >= static_cast<int>( opts.size() ) ) {
1096 return false;
1097 }
1098 auto &sel = opts[ idx ];
1099 switch( cur_key ) {
1100 case KEY_LEFT:
1101 if( can_partial_reload ) {
1102 sel.qty( sel.qty() - 1 );
1103 menu->entries[ idx ].txt = draw_row( idx );
1104 }
1105 return true;
1106
1107 case KEY_RIGHT:
1108 if( can_partial_reload ) {
1109 sel.qty( sel.qty() + 1 );
1110 menu->entries[ idx ].txt = draw_row( idx );
1111 }
1112 return true;
1113 }
1114 return false;
1115 }
1116 } cb( opts, draw_row, last_key, default_to, !base.has_flag( "RELOAD_ONE" ) );
1117 menu.callback = &cb;
1118
1119 menu.query();
1120 if( menu.ret < 0 || static_cast<size_t>( menu.ret ) >= opts.size() ) {
1121 add_msg_if_player( m_info, _( "Never mind." ) );
1122 return item::reload_option();
1123 }
1124
1125 const item_location &sel = opts[ menu.ret ].ammo;
1126 uistate.lastreload[ ammotype( base.ammo_default().str() ) ] = sel->is_ammo_container() ?
1127 sel->contents.front().typeId() :
1128 sel->typeId();
1129 return opts[ menu.ret ];
1130}

1133 bool include_empty_mags, bool include_potential ) const
1134{
1135 auto opts = base.gunmods();
1136 opts.push_back( &base );
1137
1138 if( base.magazine_current() ) {
1139 opts.push_back( base.magazine_current() );
1140 }
1141
1142 for( const auto mod : base.gunmods() ) {
1143 if( mod->magazine_current() ) {
1144 opts.push_back( mod->magazine_current() );
1145 }
1146 }
1147
1148 bool ammo_match_found = false;
1149 int ammo_search_range = is_mounted() ? -1 : 1;
1150 for( const auto e : opts ) {
1151 for( const item_location &ammo : find_ammo( *e, include_empty_mags, ammo_search_range ) ) {
1152 // don't try to unload frozen liquids
1153 if( ammo->is_watertight_container() && ammo->contents_made_of( SOLID ) ) {
1154 continue;
1155 }
1156 auto id = ( ammo->is_ammo_container() || ammo->is_container() )
1157 ? ammo->contents.front().typeId()
1158 : ammo->typeId();
1159 const bool can_reload_with = e->can_reload_with( id );
1160 if( can_reload_with ) {
1161 // Speedloaders require an empty target.
1162 if( include_potential || !ammo->has_flag( "SPEEDLOADER" ) || e->ammo_remaining() < 1 ) {
1163 ammo_match_found = true;
1164 }
1165 }
1166 if( ( include_potential && can_reload_with )
1167 || can_reload( *e, id ) || e->has_flag( "RELOAD_AND_SHOOT" ) ) {
1168 ammo_list.emplace_back( this, e, &base, ammo );
1169 }
1170 }
1171 }
1172 return ammo_match_found;

1175item::reload_option player::select_ammo( const item &base, bool prompt,
1176 bool include_empty_mags, bool include_potential ) const
1177{
1178 std::vector<item::reload_option> ammo_list;
1179 const bool ammo_match_found = list_ammo( base, ammo_list, include_empty_mags, include_potential );
1180
1181 if( ammo_list.empty() ) {
1182 if( !is_npc() ) {
1183 if( !base.is_magazine() && !base.magazine_integral() && !base.magazine_current() ) {
1184 add_msg_if_player( m_info, _( "You need a compatible magazine to reload the %s!" ),
1185 base.tname() );
1186
1187 } else if( ammo_match_found ) {
1188 add_msg_if_player( m_info, _( "Nothing to reload!" ) );
1189 } else {
1190 std::string name;
1191 if( base.ammo_data() ) {
1192 name = base.ammo_data()->nname( 1 );
1193 } else if( base.is_watertight_container() ) {
1194 name = base.is_container_empty() ? "liquid" : base.contents.front().tname();
1195 } else {
1196 name = enumerate_as_string( base.ammo_types().begin(),
1197 base.ammo_types().end(), []( const ammotype & at ) {
1198 return at->name();
1199 }, enumeration_conjunction::none );
1200 }
1201 add_msg_if_player( m_info, _( "You don't have any %s to reload your %s!" ),
1202 name, base.tname() );
1203 }
1204 }
1205 return item::reload_option();
1206 }
1207
1208 // sort in order of move cost (ascending), then remaining ammo (descending) with empty magazines always last
1209 std::stable_sort( ammo_list.begin(), ammo_list.end(), []( const item::reload_option & lhs,
1210 const item::reload_option & rhs ) {
1211 return lhs.ammo->ammo_remaining() > rhs.ammo->ammo_remaining();
1212 } );
1213 std::stable_sort( ammo_list.begin(), ammo_list.end(), []( const item::reload_option & lhs,
1214 const item::reload_option & rhs ) {
1215 return lhs.moves() < rhs.moves();
1216 } );
1217 std::stable_sort( ammo_list.begin(), ammo_list.end(), []( const item::reload_option & lhs,
1218 const item::reload_option & rhs ) {
1219 return ( lhs.ammo->ammo_remaining() != 0 ) > ( rhs.ammo->ammo_remaining() != 0 );
1220 } );
1221
1222 if( is_npc() ) {
1223 return ammo_list[ 0 ];
1224 }
1225
1226 if( !prompt && ammo_list.size() == 1 ) {
1227 // unconditionally suppress the prompt if there's only one option
1228 return ammo_list[ 0 ];
1229 }
1230
1231 return select_ammo( base, std::move( ammo_list ) );
1232}

1553int player::item_reload_cost( const item &it, const item &ammo, int qty ) const
1554{
1555 if( ammo.is_ammo() ) {
1556 qty = std::max( std::min( ammo.charges, qty ), 1 );
1557 } else if( ammo.is_ammo_container() || ammo.is_container() ) {
1558 qty = clamp( qty, ammo.contents.front().charges, 1 );
1559 } else if( ammo.is_magazine() ) {
1560 qty = 1;
1561 } else {
1562 debugmsg( "cannot determine reload cost as %s is neither ammo or magazine", ammo.tname() );
1563 return 0;
1564 }
1565
1566 // If necessary create duplicate with appropriate number of charges
1567 item obj = ammo;
1568 obj = obj.split( qty );
1569 if( obj.is_null() ) {
1570 obj = ammo;
1571 }
1572 // No base cost for handling ammo - that's already included in obtain cost
1573 // We have the ammo in our hands right now
1574 int mv = item_handling_cost( obj, true, 0 );
1575
1576 if( ammo.has_flag( "MAG_BULKY" ) ) {
1577 mv *= 1.5; // bulky magazines take longer to insert
1578 }
1579
1580 if( !it.is_gun() && !it.is_magazine() ) {
1581 return mv + 100; // reload a tool or sealable container
1582 }
1583
1584 /** @EFFECT_GUN decreases the time taken to reload a magazine */
1585 /** @EFFECT_PISTOL decreases time taken to reload a pistol */
1586 /** @EFFECT_SMG decreases time taken to reload an SMG */
1587 /** @EFFECT_RIFLE decreases time taken to reload a rifle */
1588 /** @EFFECT_SHOTGUN decreases time taken to reload a shotgun */
1589 /** @EFFECT_LAUNCHER decreases time taken to reload a launcher */
1590
1591 int cost = ( it.is_gun() ? it.get_reload_time() : it.type->magazine->reload_time ) * qty;
1592
1593 skill_id sk = it.is_gun() ? it.type->gun->skill_used : skill_gun;
1594 mv += cost / ( 1.0f + std::min( get_skill_level( sk ) * 0.1f, 1.0f ) );
1595
1596 if( it.has_flag( "STR_RELOAD" ) ) {
1597 /** @EFFECT_STR reduces reload time of some weapons */
1598 mv -= get_str() * 20;
1599 }
1600
1601 return std::max( mv, 25 );
1602}

1766bool player::unload( item_location loc )
1767{
1768 item &it = *loc.get_item();
1769 // Unload a container consuming moves per item successfully removed
1770 if( it.is_container() || it.is_bandolier() ) {
1771 if( it.contents.empty() ) {
1772 add_msg( m_info, _( "The %s is already empty!" ), it.tname() );
1773 return false;
1774 }
1775 if( !it.can_unload_liquid() ) {
1776 add_msg( m_info, _( "The liquid can't be unloaded in its current state!" ) );
1777 return false;
1778 }
1779
1780 bool changed = false;
1781 for( item *contained : it.contents.all_items_top() ) {
1782 int old_charges = contained->charges;
1783 const bool consumed = this->add_or_drop_with_msg( *contained, true );
1784 changed = changed || consumed || contained->charges != old_charges;
1785 if( consumed ) {
1786 this->mod_moves( -this->item_handling_cost( *contained ) );
1787 it.remove_item( *contained );
1788 }
1789 }
1790
1791 if( changed ) {
1792 it.on_contents_changed();
1793 }
1794 return true;
1795 }
1796
1797 // If item can be unloaded more than once (currently only guns) prompt user to choose
1798 std::vector<std::string> msgs( 1, it.tname() );
1799 std::vector<item *> opts( 1, &it );
1800
1801 for( auto e : it.gunmods() ) {
1802 if( e->is_gun() && !e->has_flag( "NO_UNLOAD" ) &&
1803 ( e->magazine_current() || e->ammo_remaining() > 0 || e->casings_count() > 0 ) ) {
1804 msgs.emplace_back( e->tname() );
1805 opts.emplace_back( e );
1806 }
1807 }
1808
1809 item *target = nullptr;
1810 if( opts.size() > 1 ) {
1811 const int ret = uilist( _( "Unload what?" ), msgs );
1812 if( ret >= 0 ) {
1813 target = opts[ret];
1814 }
1815 } else {
1816 target = &it;
1817 }
1818
1819 if( target == nullptr ) {
1820 return false;
1821 }
1822
1823 // Next check for any reasons why the item cannot be unloaded
1824 if( target->ammo_types().empty() || target->ammo_capacity() <= 0 ) {
1825 add_msg( m_info, _( "You can't unload a %s!" ), target->tname() );
1826 return false;
1827 }
1828

2106std::pair<int, int> player::gunmod_installation_odds( const item &gun, const item &mod ) const
2107{
2108 // Mods with INSTALL_DIFFICULT have a chance to fail, potentially damaging the gun
2109 if( !mod.has_flag( "INSTALL_DIFFICULT" ) || has_trait( trait_DEBUG_HS ) ) {
2110 return std::make_pair( 100, 0 );
2111 }
2112
2113 int roll = 100; // chance of success (%)
2114 int risk = 0; // chance of failure (%)
2115 int chances = 1; // start with 1 in 6 (~17% chance)
2116
2117 for( const auto &e : mod.type->min_skills ) {
2118 // gain an additional chance for every level above the minimum requirement
2119 skill_id sk = e.first == skill_weapon ? gun.gun_skill() : e.first;
2120 chances += std::max( get_skill_level( sk ) - e.second, 0 );
2121 }
2122 // cap success from skill alone to 1 in 5 (~83% chance)
2123 roll = std::min( static_cast<double>( chances ), 5.0 ) / 6.0 * 100;
2124 // focus is either a penalty or bonus of at most +/-10%
2125 roll += ( std::min( std::max( focus_pool, 140 ), 60 ) - 100 ) / 4;
2126 // dexterity and intelligence give +/-2% for each point above or below 12
2127 roll += ( get_dex() - 12 ) * 2;
2128 roll += ( get_int() - 12 ) * 2;
2129 // each level of damage to the base gun reduces success by 10%
2130 roll -= std::max( gun.damage_level( 4 ), 0 ) * 10;
2131 roll = std::min( std::max( roll, 0 ), 100 );
2132
2133 // risk of causing damage on failure increases with less durable guns
2134 risk = ( 100 - roll ) * ( ( 10.0 - std::min( gun.type->gun->durability, 9 ) ) / 10.0 );
2135
2136 return std::make_pair( roll, risk );
2137}
