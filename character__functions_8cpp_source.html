29time_duration estimate_effect_dur( int skill_lvl, const efftype_id &target_effect,
30 const time_duration &error_magnitude, int threshold, const Creature &target )
31{
32 const time_duration zero_duration = 0_turns;
33
34 time_duration estimate = std::max( zero_duration, target.get_effect_dur( target_effect ) +
35 rng( -1, 1 ) * error_magnitude *
36 rng( 0, std::max( 0, threshold - skill_lvl ) ) );
37 return estimate;
38}

74int get_book_fun_for( const Character &ch, const item &book )
75{
76 int fun_bonus = book.type->book->fun;
77 if( !book.is_book() ) {
78 debugmsg( "called avatar::book_fun_for with non-book" );
79 return 0;
80 }
81
82 // If you don't have a problem with eating humans, To Serve Man becomes rewarding
83 if( ( ch.has_trait( trait_CANNIBAL ) ||
84 ch.has_trait( trait_PSYCHOPATH ) ||
85 ch.has_trait( trait_SAPIOVORE ) ) &&
86 book.typeId() == itype_cookbook_human ) {
87 fun_bonus = std::abs( fun_bonus );
88 } else if( ch.has_trait( trait_SPIRITUAL ) && book.has_flag( "INSPIRATIONAL" ) ) {
89 fun_bonus = std::abs( fun_bonus * 3 );
90 }
91
92 if( ch.has_trait( trait_LOVES_BOOKS ) ) {
93 fun_bonus++;
94 }
95
96 if( fun_bonus > 1 && book.get_chapters() > 0 && book.get_remaining_chapters( ch ) == 0 ) {
97 fun_bonus /= 2;
98 }
99
100 return fun_bonus;
101}

108float fine_detail_vision_mod( const Character &who, const tripoint &p )
109{
110 // PER_SLIME_OK implies you can get enough eyes around the bile
111 // that you can generally see. There still will be the haze, but
112 // it's annoying rather than limiting.
113 if( who.is_blind() ||
114 ( ( who.has_effect( effect_boomered ) || who.has_effect( effect_darkness ) ) &&
115 !who.has_trait( trait_PER_SLIME_OK ) ) ) {
116 return 11.0;
117 }
118 // Scale linearly as light level approaches LIGHT_AMBIENT_LIT.
119 // If we're actually a source of light, assume we can direct it where we need it.
120 // Therefore give a hefty bonus relative to ambient light.
121 float own_light = std::max( 1.0f, LIGHT_AMBIENT_LIT - who.active_light() - 2.0f );
122
123 // Same calculation as above, but with a result 3 lower.
124 float ambient_light = std::max( 1.0f, LIGHT_AMBIENT_LIT - get_map().ambient_light_at( p ) + 1.0f );
125
126 return std::min( own_light, ambient_light );
127}

144stat_mod get_pain_penalty( const Character &ch )
145{
146 stat_mod ret;
147 int pain = ch.get_perceived_pain();
148 if( pain <= 0 ) {
149 return ret;
150 }
151
152 int stat_penalty = std::floor( std::pow( pain, 0.8f ) / 10.0f );
153
154 bool ceno = ch.has_trait( trait_CENOBITE );
155 if( !ceno ) {
156 ret.strength = stat_penalty;
157 ret.dexterity = stat_penalty;
158 }
159
160 if( !ch.has_trait( trait_INT_SLIME ) ) {
161 ret.intelligence = stat_penalty;
162 } else {
163 ret.intelligence = pain / 5;
164 }
165
166 ret.perception = stat_penalty * 2 / 3;
167
168 ret.speed = std::pow( pain, 0.7f );
169 if( ceno ) {
170 ret.speed /= 2;
171 }
172
173 ret.speed = std::min( ret.speed, 30 );
174 return ret;
175}
